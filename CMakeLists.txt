cmake_minimum_required(VERSION 3.26)
project(AA_Sidechain_Perturbation_CPP)

set(GCC_COVERAGE_COMPILE_FLAGS "-Wall -Wextra -fsanitize=address -O2 ")
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} ${GCC_COVERAGE_COMPILE_FLAGS}" )
set(CMAKE_CXX_STANDARD 26)

include_directories(include)

include(FetchContent)
FetchContent_Declare(
        argparse
        GIT_REPOSITORY https://github.com/p-ranav/argparse.git
)
FetchContent_MakeAvailable(argparse)

FetchContent_Declare(
        googletest
        URL https://github.com/google/googletest/archive/03597a01ee50ed33e9dfd640b249b4be3799d395.zip
)
# For Windows: Prevent overriding the parent project's compiler/linker settings
set(gtest_force_shared_crt ON CACHE BOOL "" FORCE)
FetchContent_MakeAvailable(googletest)


add_executable(AA_Sidechain_Perturbation_CPP
        include/geometry.h
        include/pdbparser.h
        include/constants.h
        src/geometry.cpp
        src/pdbparser.cpp
        src/main.cpp
        #src/mutation.cpp
        #include/mutation.h
        include/io.h
        src/io.cpp
        include/montecarlo.h
        src/montecarlo.cpp
        include/molecules.h
        src/molecules.cpp
        test/test.cpp
)
target_link_libraries(AA_Sidechain_Perturbation_CPP argparse gtest)
