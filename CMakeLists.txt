cmake_minimum_required(VERSION 3.22)


project(AAPerturb VERSION 1.2.0)
set(GCC_COVERAGE_COMPILE_FLAGS "-Wall -Wextra -fsanitize=address -O2 -Wsign-conversion -Wconversion -pedantic")
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} ${GCC_COVERAGE_COMPILE_FLAGS}" )
set(CMAKE_CXX_STANDARD_REQUIRED  ON)
set(CMAKE_CXX_STANDARD 23)

include_directories(include)

include(FetchContent)
FetchContent_Declare(
        argparse
        GIT_REPOSITORY https://github.com/p-ranav/argparse.git
)
FetchContent_MakeAvailable(argparse)

#FetchContent_Declare(

#)
# For Windows: Prevent overriding the parent project's compiler/linker settings
#set(gtest_force_shared_crt ON CACHE BOOL "" FORCE)
#FetchContent_MakeAvailable(googletest)


add_subdirectory(Submodules/spdlog)

if(NOT TARGET spdlog)
    # Stand-alone build
    find_package(spdlog REQUIRED)
endif()



add_executable(aaperturb
        include/geometry.h
        include/constants.h
        src/geometry.cpp
        src/main.cpp
        include/IOutils.h
        src/IOutils.cpp
        include/molecules.h
        src/molecules.cpp
        include/fancy.h
        src/fancy.cpp
        include/threadpool.h
        include/perturbrun.h
        src/perturbrun.cpp
        src/angles.cpp
        include/angles.h
)

add_executable(find_interfaces
        src/find_interfaces.cpp
        include/geometry.h
        src/angles.cpp
        include/angles.h
        include/constants.h
        src/geometry.cpp
        include/IOutils.h
        src/IOutils.cpp
        include/molecules.h
        src/molecules.cpp
        src/fancy.cpp
        include/fancy.h
)



target_link_libraries(aaperturb argparse spdlog::spdlog $<$<BOOL:${MINGW}>:ws2_32>)
target_link_libraries(find_interfaces argparse spdlog::spdlog $<$<BOOL:${MINGW}>:ws2_32>)


install(TARGETS aaperturb DESTINATION ${CMAKE_INSTALL_PREFIX})
install(TARGETS find_interfaces DESTINATION ${CMAKE_INSTALL_PREFIX})
